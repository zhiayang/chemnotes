									\hbox{
										\vspace{5mm}
										{#2}
									}
								}
								\end{center}
								\begin{center}
								\fontsize{11pt}{13pt}\selectfont{#3}
								\normtext\end{center}\end{figure}
								\vspace{-5mm}
							}
